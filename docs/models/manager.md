# Manager

## Example Usage

```typescript
import { Manager } from "@kombo-api/sdk/models";

let value: Manager = {
  first_name: "John",
  last_name: "Doe",
  display_full_name: "John Doe",
  id: "26vafvWSRmbhNcxJYqjCzuJg",
  work_email: "john.doe@acme.com",
  remote_id: "32",
  employment_status: "INACTIVE",
  termination_date: new Date("2022-05-20T00:00:00Z"),
};
```
